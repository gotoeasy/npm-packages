## Install
```
npm i @gotoeasy/code-frame
```

## API
```js
const codeframe = require('@gotoeasy/code-frame');

let result;
// 传入文本、位置
result = codeframe(text, index);

// 传入文本、开始位置、结束位置
result = codeframe(text, startIndex, endIndex);

// 传入文件、行、列
result = codeframe(file, line, column);


```
